using Appeon.ApiDoc.Models.School;
using Appeon.Data;
using Microsoft.AspNetCore.Mvc;

namespace Appeon.ApiDoc.ModelStoreExamples
{
    // [JSON: Jsons\ModelStore\ModelStoreExample.json]

    [Route("examples/IModelEntry/[action]")]
    public class ModelStoreExampleController : Controller
    {
        SchoolContext _dataContext;
        public ModelStoreExampleController(SchoolContext dataContext)
        {
            // Sets the data context.
            _dataContext = dataContext;
        }

        [HttpPost]
        public string Example(IDataUnpacker dataUnpacker)
        {
            // Uses IDataUnpack to receive JSON data from the PB client app.
            var student =
                dataUnpacker.GetModelEntries<Person>("student");

            var example = new ModelStoreExample(_dataContext);

            example.Example4(student);

            return "SUCCESS";
        }
    }
}
