IGroupingGetter Interface

.NET Standard 2.x | Current Version (0.5.0-alpha)

0.5.0-alpha 0.5.1-alpha(current)
表示一个分组的获取器。与DataStore的分组功能一块使用。

当IsGroup属性为true时，用于得到一个分组的数据；当IsGroup属性为false时，用于得到某一行的数据。

Represent the getter for a group. Used in conjunction with the grouping function of the DataStore.

When the IsGroup property is true, it is used to get a group of data; when the IsGroup property is false, it is used to get data of a certain row.

using Appeon.Data.PowerBuilder;
using System;

namespace Appeon.ApiDoc.IGroupingGetterExamples
{
    public class IGroupingGetterExample
    {
        private readonly SchoolContext _context;

        public IGroupingGetterExample(SchoolContext context)
        {
            _context = context;
        }

        public void Example()
        {
            // 创建一个datastore对象
            IDataStore course = new DataStore("d_department_course", _context);

            course.Retrieve();
            
            var departmentGropup = course.GroupBy();

            // 获取分组信息
            foreach(var m in departmentGropup)
            {
                Console.WriteLine(m.Key);
                // 循环输出每个分组内的信息
                foreach (var getter in m)
                {
                    Console.WriteLine(getter.GetItemString(0)); 
                }
                
            }

            /*This code produces the following output:
            
            
            */
        }
    }
}
