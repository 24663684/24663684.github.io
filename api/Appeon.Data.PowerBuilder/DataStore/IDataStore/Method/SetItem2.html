IDataStore.SetItem(int row, string column, object value, DwBuffer bufferType = DwBuffer.Primary) Method

.NET Standard 2.x | Current Version(0.5.0-alpha)

0.5.0-alpha

0.5.1-alpha(current)

Sets the value of a row and column (by column number) in the DataStore to the specified value.

Namespace: Appeon.Data.PowerBuilder

Assembly: Appeon.Data.PowerBuilder.dll

Syntax

  public bool SetItem(int row, string column, object value, DwBuffer bufferType = DwBuffer.Primary);

Parameters

row System.Int32

The zero-based row location of the data.

column System.String

The column location (specified by a zero-based column number) of the data.

value System.Object

The value to which you want to set the data at the row and column location. The datatype of the value must be the same as the datatype of the column.

bufferType Appeon.Data.PowerBuilder.DwBuffer

The specified buffer of the DataStore.

The default is DwBuffer.Primary.

Returns

System.Boolean

Returns true if it succeeds.

Examples

The following code example uses SetItem to modify the value for the name column in the last row.

using Appeon.Data.PowerBuilder;
using System;

namespace Appeon.ApiDoc.IDataStoreExamples
{
    public class SetItemExample
    {
        private SchoolContext _context;

        public SetItemExample(SchoolContext dataContext)
        {
            // Sets the data context
            _context = dataContext;
        }

        public void Example2()
        {
            // Instantiates datastore with datawindow: d_department
            var datastore = new DataStore("d_department", _context);

            // Retrieves rows from the database for datastore
            datastore.Retrieve();

            Console.WriteLine("After Retrieve, Rowcount: {0}",
                              datastore.RowCount);

            // Sets the value for the name column in the last row to "Chinese"
            datastore.SetItem(datastore.RowCount - 1, "name", "Chinese", 
                              DwBuffer.Primary);

            Console.WriteLine("Gets the original value of name: {0}",
                               datastore.GetItemString(datastore.RowCount - 1, 
                               "name", DwBuffer.Primary, true));

            Console.WriteLine("Gets the modified value of name: {0}",
                               datastore.GetItemString(datastore.RowCount - 1,
                               "name", DwBuffer.Primary, false));

            /*This code produces the following output:

            After Retrieve, Rowcount: 4   
            Gets the original value of name: Mathematics
            Gets the modified value of name: Chinese
            */
        }
    }
}
