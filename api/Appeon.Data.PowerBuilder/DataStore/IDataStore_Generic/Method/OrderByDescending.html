IDataStore.OrderByDescending<TModel>(Func<TModel, object> keySelector) Method

按照某个key对DataStore的主缓冲区的行进行降序排序。

Sorts the rows of the primary buffer in the DataStore in descending order according to a key.

Namespace: Appeon.Data.PowerBuilder

Assembly: Appeon.Data.PowerBuilder.dll

Syntax

  public IOrderable<TModel> OrderByDescending<TModel>(Func<TModel, object> keySelector);

Examples

OrderBy(Func<TModel, object> keySelector) 使datastore 按某个条件降序排列

using Appeon.Data.PowerBuilder;
using System;

namespace Appeon.ApiDoc.IDataStoreExamples
{
    public class OrderByDescendingExample
    {
        private SchoolContext _context;

        public OrderByDescendingExample(SchoolContext dataContext)
        {
            // Set Data Context
            _context = dataContext;
        }

        public void Example()
        {
            // Instantiate the datastore with datawindow:d_department
            var datastore = new DataStore("d_department", _context);

            datastore.Retrieve();

            Console.WriteLine("Before OrderBy");

            for (int row = 0, rowcount = datastore.RowCount; row < rowcount; row++)
            {
                Console.WriteLine("Department ID:{0}; Department Name:{1}; Budget:{2}",
                               datastore.GetItem<int>(row, 0),
                               datastore.GetItem<string>(row, 1),
                               datastore.GetItem<decimal>(row, 2));
            }

            // 使datastore按department budget 降序排列
            datastore.OrderByDescending<D_Department>(d => d.Budget).Sort();

            Console.WriteLine("After OrderBy");

            for (int row = 0, rowcount = datastore.RowCount; row < rowcount; row++)
            {
                Console.WriteLine("Department ID:{0}; Department Name:{1}; Budget:{2}",
                               datastore.GetItem<int>(row, 0),
                               datastore.GetItem<string>(row, 1),
                               datastore.GetItem<decimal>(row, 2));
            }

            /*This code produces the following output:
             
            Before OrderBy
            Department ID:1; Department Name:Engineering; Budget:350000.0000
            Department ID:2; Department Name:English; Budget:120000.0000
            Department ID:4; Department Name:Economics; Budget:200000.0000
            Department ID:7; Department Name:Mathematics; Budget:250000.0000
            After OrderBy
            Department ID:1; Department Name:Engineering; Budget:350000.0000
            Department ID:7; Department Name:Mathematics; Budget:250000.0000
            Department ID:4; Department Name:Economics; Budget:200000.0000
            Department ID:2; Department Name:English; Budget:120000.0000
            */
        }
    }
}
