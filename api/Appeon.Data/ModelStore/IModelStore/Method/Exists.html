IModelStore<TModel>.Exists(Predicate<TModel> match) Method

.NET Standard 2.x | Current Version (0.5.0-alpha)

0.5.0-alpha

0.5.1-alpha (current)

Determines whether ModelStore contains the model object that matches the condition defined by the predicate.

Namespace: Appeon.Data

Assembly: Appeon.Data.dll

Syntax

  public bool Exists(Predicate<TModel> match);

Parameters

match System.Predicate<TModel>

The Predicate<TModel> object that defines the condition.

Returns

System.Boolean

Returns true if there is a model object in ModelStore that satisfies the specified condition, otherwise false.

Examples

The following code example demonstrates how to use the Exists method to check whether there is a model object in ModelStore that meets the specified condition.

using Appeon.Data;
using Appeon.ApiDoc.Models.School;
using System;

namespace Appeon.ApiDoc.IModelStoreExamples
{
    public class ExistsExample
    {
        private readonly SchoolContext _context;

        public ExistsExample(SchoolContext dataContext)
        {
            // Sets the data context
            _context = dataContext;
        }

        public void Example()
        {
            // Creates a ModelStore and retrieves data.
            IModelStore<Department> dept = new ModelStore<Department>();
            dept.Retrieve(_context);

            // Checks if the model object (DepartmentID = 1) exists.
            bool isExists = dept.Exists(m => m.DepartmentID == 1);
            Console.WriteLine("Existence (DepartmentID = 1): {0}", isExists);

            // Checks if the the model object (DepartmentID = 100) exists.
            isExists = dept.Exists(m => m.DepartmentID == 100);
            Console.WriteLine("Existence (DepartmentID = 100) {0}", isExists);

            /* This code example produces the following output:
            
            Existence (DepartmentID = 1): True
            Existence (DepartmentID = 100) False
            */
        }
    }
}
