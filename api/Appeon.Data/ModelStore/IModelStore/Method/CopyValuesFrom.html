IModelStore<TModel>.CopyValuesFrom(int index, TModel model)Method

.NET Standard 2.x | Current Version (0.5.0-alpha)

0.5.0-alpha

0.5.1-alpha (current)

Copies the property values from an external model object to the current ModelStore.

Namespace: Appeon.Data

Assembly: Appeon.Data.dll

Syntax

  public void CopyValuesFrom(int index, TModel model);

Parameters

index System.Int32

A zero-based index of ModelStore to which data values will be copied.

model TModel

A model object from which data will be copied.

Examples

The following code example demonstrates how to use the CopyValuesFrom method to copy the data from an external model object to an internal model object at the specified position in ModelStore.

using Appeon.Data;
using Appeon.ApiDoc.Models.School;
using System;

namespace Appeon.ApiDoc.IModelStoreExamples
{
    public class CopyValuesFromExample
    {
        private readonly SchoolContext _context;

        public CopyValuesFromExample(SchoolContext dataContext)
        {
            // Sets the data context
            _context = dataContext;
        }

        public void Example()
        {
            // Creates the "courseNew" model object.
            var courseNew = new Course()
            {
                CourseID = 5000,
                Title = "New Course",
                Credits = 4,
                DepartmentID = 4
            };

            // Creates a ModelStore and retrieves data.
            var courses = new ModelStore<Course>();
            courses.Retrieve(_context);

            // Displays the original values of the model object at index 1 in ModelStore.
            var course = courses.Get(1);
            Console.WriteLine("CourseID (Orginal): {0}", course.CourseID);
            Console.WriteLine("Title (Orginal): {0}", course.Title);
            Console.WriteLine("Credits (Orginal): {0}", course.Credits);
            Console.WriteLine("DepartmentID (Orginal): {0}", course.DepartmentID);

            // Copies the values from courseNew to the model object 
            // at index 1 in ModelStore.
            courses.CopyValuesFrom(1, courseNew);

            // Displays the new values of the model object at index 1 in ModelStore.
            course = courses.Get(1);
            Console.WriteLine();
            Console.WriteLine("CourseID (current): {0}", course.CourseID);
            Console.WriteLine("Title (current): {0}", course.Title);
            Console.WriteLine("Credits (current): {0}", course.Credits);
            Console.WriteLine("DepartmentID (current): {0}", course.DepartmentID);

            /* This code produces the following output:

            CourseID (Orginal): 1050
            Title (Orginal): Chemistry
            Credits (Orginal): 4
            DepartmentID (Orginal): 1

            CourseID (current): 5000
            Title (current): New Course
            Credits (current): 4
            DepartmentID (current): 4
            */
        }
    }
}
