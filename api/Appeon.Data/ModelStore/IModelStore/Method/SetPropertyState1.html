IModelStore<TModel>.SetPropertyState(int index, int propertyIndex, PropertyState propertyState) Method

.NET Standard 2.x | Current Version (0.5.0-alpha)

0.5.0-alpha

0.5.1-alpha (current)

Sets the state for the specified property (by index) of the specified model object (by index) in ModelStore, when ModelStore is being tracked using the ChangeTrackingStrategy.PropertyState strategy.

Namespace: Appeon.Data

Assembly: Appeon.Data.dll

Syntax

  public void SetPropertyState(int index, int propertyIndex, PropertyState propertyState);

Parameters

index System.Int32

The zero-based index of the model object in ModelStore.

propertyIndex System.Int32

The zero-based index of the property to which you will set state.

propertyState Appeon.Data.PropertyState

A value of the PropertyState enumerated datatype.

Examples

The following code example demonstrates how to set the state for the property (specified by index) of the record in ModelStore.

using Appeon.Data;
using Appeon.ApiDoc.Models.School;
using System;

namespace Appeon.ApiDoc.IModelStoreExamples
{
    public class SetPropertyStateExample
    {
        private readonly SchoolContext _context;

        public SetPropertyStateExample(SchoolContext dataContext)
        {
            // Sets the data context
            _context = dataContext;
        }

        public void Example1()
        {
            var courses =
                new ModelStore<Course>(ChangeTrackingStrategy.PropertyState);

            // Retrieves data.
            courses.Retrieve(_context);

            // Shows the state of the CourseID property (at index 0) of 
            // the model object (at index 0).
            var state = courses.GetPropertyState(0, 0);
            Console.WriteLine("Property State (Before SetPropertyState): {0}",
                state);

            // Shows the model state before modified.
            Console.WriteLine("Model State (Before SetPropertyState): {0}",
                courses.GetModelState(0));

            // Modifies the state of the CourseID property.
            courses.SetPropertyState(0, 0, PropertyState.Modified);

            // Shows the state after modified
            state = courses.GetPropertyState(0, 0);
            Console.WriteLine();
            Console.WriteLine("Property State (After SetPropertyState): {0}",
                state);

            // Shows the model state after modified.
            Console.WriteLine("Model State (After SetPropertyState): {0}",
                courses.GetModelState(0));

            /* This code example produces the following output:
            
            Property State (Before SetPropertyState): NotModified
            Model State (Before SetPropertyState): NotTracked

            Property State (After SetPropertyState): Modified
            Model State (After SetPropertyState): Modified
            */
        }
    }
}
