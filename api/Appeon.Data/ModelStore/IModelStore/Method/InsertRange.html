IModelStore<TModel>.InsertRange(int index, IEnumerable<TModel> models) Method

.NET Standard 2.x | Current Version (0.5.0-alpha)

0.5.0-alpha

0.5.1-alpha (current)

Inserts a sequence of model objects to the specified position in ModelStore.

Namespace: Appeon.Data

Assembly: Appeon.Data.dll

Syntax

  public void InsertRange(int index, IEnumerable<TModel> models);

Parameters

index System.Int32

The zero-based index at which model objects should be inserted.

models System.Collections.Generic.IEnumerable<TModel>

A sequence of model objects to be inserted.

Remarks

If ModelStore is being tracked, the newly inserted model objects will have the ModelState.NewModified state.

Examples

The following code example demonstrates how to use the InsertRange method to insert multiple model objects to ModelStore.

using Appeon.Data;
using Appeon.ApiDoc.Models.School;
using System;
using System.Collections.Generic;

namespace Appeon.ApiDoc.IModelStoreExamples
{
    public class InsertRangeExample
    {
        private readonly SchoolContext _context;

        public InsertRangeExample(SchoolContext dataContext)
        {
            // Sets the data context
            _context = dataContext;
        }

        public void Example()
        {
            var courses = new ModelStore<Course>();

            // Retrieves data.
            courses.Retrieve(_context);

            // Creates the Course objects.
            var courseRange = new List<Course>();
            Course course = new Course()
            {
                CourseID = 1,
                Title = "New Course 1",
                Credits = 1,
                DepartmentID = 1
            };

            Course course2 = new Course()
            {
                CourseID = 2,
                Title = "New Course 2",
                Credits = 1,
                DepartmentID = 1
            };

            // Adds two Course records to List
            courseRange.Add(course);
            courseRange.Add(course2);

            // Shows the number of records in ModelStore before insertion
            Console.WriteLine("Before Insert Count: {0}", courses.Count);

            // Inserts the data record before row 2
            courses.InsertRange(1, courseRange);

            // Shows the number of records in ModelStore after insertion
            Console.WriteLine("After Insert Count: {0}", courses.Count);

            /* This code example produces the following output:
            
             Before Insert Count: 11
             After Insert Count: 13
            */
        }
    }
}
