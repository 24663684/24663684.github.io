IModelStore<TModel>.Validate() Method

.NET Standard 2.x | Current Version (0.5.0-alpha)

0.5.0-alpha

0.5.1-alpha (current)

Validates the data of the model objects in ModelStore using the validation rules set in TModel.

Namespace: Appeon.Data

Assembly: Appeon.Data.dll

Syntax

  public bool Validate();

Returns

System.Boolean

Returns true if all data meets the validation rules; false if any data does not meet the validation rules.

Examples

The following code example demonstrates how to perform validity check to the data in ModelStore.

using Appeon.Data;
using Appeon.ApiDoc.Models.School;
using System;
using System.Collections.Generic;
using System.ComponentModel.DataAnnotations;
using System.Linq;

namespace Appeon.ApiDoc.IModelStoreExamples
{
    public class ValidateExample
    {
        private readonly SchoolContext _context;

        public ValidateExample(SchoolContext dataContext)
        {
            // Sets the data context
            _context = dataContext;
        }

        public void Example1()
        {
            // Credits is not allowed to be less than 2 or greater than 5.
            var courses = new ModelStore<CourseValidation>(
                ChangeTrackingStrategy.PropertyState);

            // Retrieves data.
            courses.Retrieve(_context);

            // Sets a valid value for Credits.
            courses.SetValue(0, "Credits", 3);

            // Validates the data.
            bool result = courses.Validate();
            Console.WriteLine("Validate: {0}.", result);

            // Sets an invalid value for Credits.
            courses.SetValue(0, "Credits", 6);

            // Validates the data.
            result = courses.Validate();
            Console.WriteLine("Validate: {0}.", result);

            /* This code example produces the following output:
            
            Validate: True.
            Validate: False.
            */
        }
    }
}
