确定ModelStore是否包含特定的model object。

Determines whether the specified model object exists in ModelStore.

Namespace: Appeon.Data

Assembly: Appeon.Data.dll

Syntax

  public bool Contains(TModel item);

Parameters

item TModel

The model object to locate in ModelStore.

Returns

System.Boolean

True if item is found in ModelStore; otherwise, false.

Remarks

Examples

The following code example demonstrates how to 查找ModelStore中是否包含指定的记录

using Appeon.Data;
using Appeon.ApiDoc.Models.School;
using System;

namespace Appeon.ApiDoc.IModelStoreExamples
{
    public class ContainsExample
    {
        private readonly SchoolContext _context;

        public ContainsExample(SchoolContext dataContext)
        {
            // Sets the data context
            _context = dataContext;
        }

        public void Example()
        {
            // 定义两条Department记录
            Department newdept = new Department()
            {
                DepartmentID = 8001,
                Name = "test department",
                Budget = 1m,
                StartDate = DateTime.Parse("2018-12-12")
            };

            Department newdept2 = new Department()
            {
                DepartmentID = 1,
                Name = "Engineering",
                Budget = 350000.00m,
                StartDate = DateTime.Parse("2007-09-01"),
                Administrator = 2
            };

            IModelStore<Department> dept = new ModelStore<Department>();

            //  Retrieves data from Department
            dept.Retrieve(_context);

            // 查找ModelStore中是否包含指定记录1
            bool find = dept.Contains(newdept);

            // 未发现匹配记录，返回false
            Console.WriteLine("Result = {0}", find);

            /*This code produces the following output:
             
              Result = False
            */

            // 查找ModelStore中是否包含指定记录2
            bool finds = dept.Contains(newdept2);

            // 发现匹配记录，返回true
            Console.WriteLine("Result = {0}", finds);

            /*This code produces the following output:
            
              Result = True
            */
        }
    }
}
