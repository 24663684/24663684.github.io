IModelStore<TModel>.Remove(TModel item) Method

.NET Standard 2.x | Current Version(0.5.0-alpha)

0.5.0-alpha

0.5.1-alpha(current)

从ModelStore中删除指定的model object。

Removes the specified model object from the ModelStore.

Namespace: Appeon.Data

Assembly: Appeon.Data.dll

Syntax

  public bool Remove(TModel item);

Parameters

item TModel

The model object to remove from the ModelStore.

Returns

System.Boolean

True if this model object has been removed;

false if not removed.

Remarks

If the ModelStore is being tracked, the removed model object will get ModelState.Deleted state.

Examples

Remove Model中指定的记录

using Appeon.Data;
using Appeon.ApiDoc.Models.School;
using System;

namespace Appeon.ApiDoc.IModelStoreExamples
{
    public class RemoveExample
    {
        private readonly SchoolContext _context;

        public RemoveExample(SchoolContext dataContext)
        {
            // Set data context
            _context = dataContext;
        }

        public void Example()
        {
            IModelStore<Course> courses = new ModelStore<Course>();

            // 检索出courses中的数据
            courses.Retrieve(_context);

            // 获取Model中的第二行记录
            Course moveCourse = courses.Get(1);

            // 输出Remove前Model中的记录行数
            Console.WriteLine("Before Remove Count = {0}", courses.Count);

            // Remove Model中该记录
            courses.Remove(moveCourse);

            // 输出Remove后Model中的记录行数
            Console.WriteLine("After Remove Count = {0}", courses.Count);

            /* This code example produces the following output:
            
             Before Remove Count = 12
             After Remove Count = 11
            */
        }
    }
}
