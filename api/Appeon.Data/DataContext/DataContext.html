Examples

The following code example demonstrates how to create a DataContextExample class which can be used to connect the connect a SQL Server database.

using Appeon.Data;
using Appeon.Data.SqlServer;
using System.Data.SqlClient;

namespace Appeon.ApiDoc.DataContextExamples
{
    public class DataContextExample : SqlServerDataContext
    {
        /// <summary>
        /// 构造DataContext类
        /// 使用DataContextOptions参数来构造
        /// </summary>              
        public DataContextExample(IDataContextOptions<DataContextExample> options)
            : base((SqlServerDataContextOptions)options)
        {

        }

        /// <summary>
        /// 构造DataContext类
        /// 使用SqlConnection参数来构造
        /// </summary>        
        public DataContextExample(SqlConnection conns)
            : base(conns)
        {

        }
    }
}
