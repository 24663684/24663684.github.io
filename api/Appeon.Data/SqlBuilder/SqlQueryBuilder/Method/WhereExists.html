Examples

判断当前表中数据是否存在子查询返回的数据中，并将查询结果返回Model中

using Appeon.Data;
using System;

namespace Appeon.ApiDoc.SqlBuilderExamples
{
    public class WhereExistsExample
    {
        private SchoolContext _context;

        public WhereExistsExample(SchoolContext dataContext)
        {
            // Set Data Context
            _context = dataContext;
        }

        public void Example()
        {
            ISqlQueryBuilder sqlquerybuilder = new SqlQueryBuilder();

            // 定义子查询
            ISqlQueryBuilder subsqlbuilder = new SqlQueryBuilder();
            subsqlbuilder.Select("1")
                .From("Department")
                .Where("departmentid", "c.departmentid");

            // 定义Where查询
            var where =
                SqlBuilder.WhereExists(subsqlbuilder);

            // 创建sql语法, 并转换成context对应数据库的可执行语句
            string sql = sqlquerybuilder
                .Select("*")
                .From("Course", "c")
                .Where(where)
                .ToSqlString(_context);

            Console.WriteLine(sql);

            // execute sql
            var departments = _context.SqlExecutor.Select<DynamicModel>(sql);

            /*This code example produces the following output:   
             
            SELECT
            *
            FROM [Course] AS [c]
            WHERE ( EXISTS (SELECT
            1
            FROM [Department]
            WHERE ([departmentid] = [c].[departmentid])))
            */
        }
    }
}
