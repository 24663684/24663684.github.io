ISqlConditionBuilder.ToSqlString(DataContext context) 方法

返回当前查找条件在相应数据库的原生SQL字符串。

Returns the raw SQL of current search condition for the corresponding database.

命名空间： Appeon.Data

程序集： Appeon.Data.dll

语法

 string ToSqlString(DataContext context);

Parameters

context Appeon.Data.DataContext

A DataContext object which contains database connection info.

Returns

System.String

Returns the raw SQL string.

备注

Examples

演示ToSqlString方法。将sqlWhereCondition语法转换成context对应数据库的可执行语句

using Appeon.Data;
using System;

namespace Appeon.ApiDoc.ISqlConditionBuilderExamples
{
    public class ToSqlStringExample
    {
        private SchoolContext _context;

        public ToSqlStringExample(SchoolContext dataContext)
        {
            // Set Data Context
            _context = dataContext;
        }

        public void Example()
        {
            // Gets a search conditon for WHERE clause.
            var sqlWhereCondition = SqlBuilder.WhereValue("DepartmentId", 1);

            // 将sqlWhereCondition语法转换成context对应数据库的可执行语句
            Console.WriteLine("The search conditon is: \n{0}",
                sqlWhereCondition.ToSqlString(_context));

            // 定义SQL语句, 使用sqlWhereCondition
            var sqlbuilder = SqlBuilder.Select("name")
                                        .From("Department")
                                        .Where(sqlWhereCondition);

            // 将sqlbuilder语法转换成context对应数据库的可执行语句
            string sql = sqlbuilder.ToSqlString(_context);

            Console.WriteLine();
            Console.WriteLine("The SQL SELECT statement: \n{0}", sql);

            /*This code produces the following output:
            
            The search conditon is:
            WHERE ([DepartmentId] = 1)

            The SQL SELECT statement:
            SELECT
             [name]
            FROM [Department]
            WHERE ([DepartmentId] = 1)           
            */
        }
    }
}
