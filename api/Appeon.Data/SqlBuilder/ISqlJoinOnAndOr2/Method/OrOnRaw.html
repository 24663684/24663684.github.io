ISqlJoinOnAndOr<TBuilder>.OrOnRaw(string clause, params ISqlParameter[] parameters) 方法

.NET Standard 2.x | 当前版本(0.5.0-alpha)

0.5.0-alpha

0.5.1-alpha(current)

向ON子句添加OR逻辑运算符和raw SQL。

Add the OR logical operator and an raw SQL to the ON clause.

命名空间： Appeon.Data

程序集： Appeon.Data.dll

语法

  TBuilder OrOnRaw(string clause, params ISqlParameter[] parameters);

Parameters

clause System.String

A string of raw SQL added to the ON clause.

parameters Appeon.Data.ISqlParameter[]

(Optional) An array of ISqlParameter objects which defines the SQL parameters used in the raw SQL.

Returns

TBuilder

Returns the TBuilder object which can be used to add more search conditions for current ON clause.

备注

Examples

一表连接另外一个表时，如果包含多个或连接条件使用OrOnRaw附加条件

using Appeon.Data;
using System;

namespace Appeon.ApiDoc.ISqlJoinOnAndOrExamples
{
    public class OrOnRawExample
    {
        private readonly SchoolContext _context;

        public OrOnRawExample(SchoolContext dataContext)
        {
            // Set data context
            _context = dataContext;
        }

        public void Example()
        {
            var sqlQueryBuilder = new SqlQueryBuilder();

            // 获取Department表中DepartmentID，Name; Course表中Title数据，
            // Department和Course表关联条件为Department.DepartmentID = Course.DepartmentID
            // 并且Department.DepartmentID 等于一个输入参数1
            // 或者Department.DepartmentID 等于一个输入参数2
            sqlQueryBuilder
                .Select("Department.DepartmentID")
                .Select("Department.Name")
                .Select("Course.Title")
                .From("Department")
                .Join("Course")
                .OnRaw("Department.DepartmentID = Course.DepartmentID")
                .AndOnRaw("(Department.DepartmentID = @departmentid1 ",
                         SqlBuilder.Parameter<int>("departmentid1"))
                .OrOnRaw("Department.DepartmentID = @departmentid2)",
                        SqlBuilder.Parameter<int>("departmentid2"));

            // 将sqlQueryBuilder语法转换成context对应数据库的可执行语句
            string sql = sqlQueryBuilder.ToSqlString(_context);

            Console.WriteLine(sql);

            // execute sql, 指定两个输入参数为1, 2
            var departments = _context.SqlExecutor.Select<DynamicModel>(sql, 1, 2);

            /*This code example produces the following output:           
              SELECT
                [Department].[DepartmentID],
                [Department].[Name],
                [Course].[Title]
              FROM [Department] JOIN [Course]
              ON [Department].[DepartmentID] = [Course].[DepartmentID]
              AND ([Department].[DepartmentID] = @departmentid1
              OR [Department].[DepartmentID] = @departmentid2)
            */
        }
    }
}
