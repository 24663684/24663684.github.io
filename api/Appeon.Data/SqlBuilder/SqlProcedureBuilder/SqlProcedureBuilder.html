Examples

创建一个SqlProcedureBuilder对象

using Appeon.Data;
using System;

namespace Appeon.ApiDoc.SqlProcedureBuilderExamples
{
    public class SqlProcedureBuilderExample
    {
        private readonly SchoolContext _context;

        public SqlProcedureBuilderExample(SchoolContext dataContext)
        {
            // Set data context
            _context = dataContext;
        }

        public void Example()
        {
            // 创建SqlProcedureBuilder对象
            ISqlProcedureBuilder proce = new SqlProcedureBuilder();

            // 使用该对象构造插入语句
            proce.Procedure("proce_test", SqlBuilder.Parameter<int>("id"));

            // 输出生成的SQL语句
            var sql = proce.ToSqlString(_context);
            Console.WriteLine(sql);

            /*This code produces the following output:
            
            INSERT INTO Course([Courseid],
            [Title]) VALUES(@id,
            @title)   
            */
        }
    }
}
