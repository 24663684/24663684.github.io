using Appeon.Data;
using Appeon.ApiDoc.Models.School;
using Microsoft.AspNetCore.Mvc;
using System.Linq;

namespace Appeon.ApiDoc.IDetailTrack
{
    [Route("examples/IDetailTrack/TrackDetail/[action]")]
    public class TrackDetailsExampleController : Controller
    {
        private readonly SchoolContext _dataContext;

        public TrackDetailsExampleController(SchoolContext dataContext)
        {
            _dataContext = dataContext;
        }

        [HttpPost]
        public string Example(IDataUnpacker dataUnpacker)
        {
            // Uses IDataUnpack to receive JSON data from the PB client app.
            // [JSON: Jsons\IDetailTrack\TrackDetailsExample]            

            // Gets the data of Course and OnlineCourse from the PB client app.
            var depart = dataUnpacker.GetModelEntries<DepartmentInfo>("course").FirstOrDefault();
            var coures = dataUnpacker.GetModelEntries<CourseInfo>("onlinecourse");

            // Saves data.
            var example = new IDetailTrackerExamples.TrackDetailsExample(_dataContext);
            example.Example(depart, coures);

            return "SUCCESS";
        }

    }
}
