Remarks

Examples

清空Model中指定StateTrackable的记录

using Appeon.Data;
using Appeon.ApiDoc.Models.School;
using System;

namespace Appeon.ApiDoc.IModelStoreExamples
{
    public class ResetStateExample
    {
        private readonly SchoolContext _context;

        public ResetStateExample(SchoolContext dataContext)
        {
            // Set data context
            _context = dataContext;
        }

        public void Example()
        {
            IModelStore<Course> courses = new ModelStore<Course>();

            // 检索出courses中的数据
            courses.Retrieve(_context);

            // 输出Reset前Model中的记录行数
            Console.WriteLine("Before Reset Count = {0}", courses.Count);

            // Remove Model中StateTrackable = Deleted的记录
            courses.ResetState( StateTrackable.Deleted);

            // 输出Reset后Model中的记录行数
            Console.WriteLine("After Reset Count = {0}", courses.Count);

            /* This code example produces the following output:
            
             Before Reset Count = 12
             After Reset Count = 11
            */
        }
    }
}
