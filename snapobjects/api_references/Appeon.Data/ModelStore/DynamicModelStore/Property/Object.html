用于从脚本直接操作DataWindow object中的对象。 例如，这些对象可以是列或文本对象。

Used for the direct manipulation of objects within a DataWindow object from a script. These objects can be, for example, columns or text objects.

Namespace: Appeon.Data.PowerBuilder

Examples

演示Object方法

using Appeon.Data.PowerBuilder;
using System;

namespace Appeon.ApiDoc.IDataStoreExamples
{
    public class ObjectExample
    {
        private SchoolContext _context;

        public ObjectExample(SchoolContext dataContext)
        {
            // Set Data Context
            _context = dataContext;
        }

        public void Example()
        {
            // Instantiate the datastore with datawindow:d_department
            var datastore = new DataStore("d_department", _context);

            // 获取datastore中列名为name的Visible属性值
            Console.WriteLine("Get the visible property of name:{0}",
                               datastore.Object.Columns["name"].Visible);

            // 获取datastore中第一个标签的name属性值
            Console.WriteLine("Get the name property of the first column label:{0}"
                               ,datastore.Object.Texts[0].Name.ToString());

            // 获取datastore中detail区域的高度值
            Console.WriteLine("Get the Height of detail band:{0}", 
                              datastore.Object.Bands[1].Height.ToString());
            /*This code produces the following output:
            
            Get the visible property of name:True
            Get the name property of the first column label:departmentid_t
            Get the Height of detail band:88   
            */

        }
    }
}
