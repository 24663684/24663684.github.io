Gets the number of the model objects in the specified model state(s).

Namespace: Appeon.Data

Assembly: Appeon.Data.dll

Syntax

  public int TrackedCount(StateTrackable stateTrackable);

Parameters

stateTrackable Appeon.Data.StateTrackable

A value of the StateTrackable enumerated datatype which specifies one or more model states.

Returns

System.Int32

Returns the number of the model objects in the specified state(s).

Examples

The following code example demonstrates how to use the TrackedCount method to calculate the number of model objects that are in the ModelState.Modified state.

using Appeon.Data;
using Appeon.ApiDoc.Models.School;
using System;

namespace Appeon.ApiDoc.IModelStoreExamples
{
    public class TrackedCountExample
    {
        private readonly SchoolContext _context;

        public TrackedCountExample(SchoolContext dataContext)
        {
            // Sets the data context
            _context = dataContext;
        }

        public void Example()
        {
            var courses =
                new ModelStore<Course>(ChangeTrackingStrategy.PropertyState);

            // Retrieves data.
            courses.Retrieve(_context);

            // Gets the number of model objects that are in the Modified state
            int count = courses.TrackedCount(StateTrackable.Modified);

            // Shows the number of model objects that are in the Modified state
            Console.WriteLine("Modified State Count (Before Modified): {0}",
                count);

            // Modifies the value.
            courses.SetValue(0, 1, "New Course");

            // Gets the number of model objects that are in the Modified state
            count = courses.TrackedCount(StateTrackable.Modified);

            // Shows the number of model objects that are in the Modified state
            Console.WriteLine("Modified State Count (After Modified): {0}",
                count);

            /* This code example produces the following output:
            
            Modified State Count (Before Modified): 0
            Modified State Count (After Modified): 1
            */
        }
    }
}
