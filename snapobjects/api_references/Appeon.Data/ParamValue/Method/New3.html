ParamValue.New<Type>(string name, ParameterDirection direction) Method

.NET Standard 2.x | Current Version(0.5.0-alpha)

0.5.0-alpha

0.5.1-alpha (current)

Creates a new ParamValue object and specifies the name and the direction type of the parameter.

新建一个ParamValue对象，指定parameterName和方向。

Namespace: Appeon.Data

Assembly: Appeon.Data.dll

Syntax

public static ParamValue New<Type>(string name, ParameterDirection direction)

Parameters

name System.String

The name of the parameter.

direction System.Data.ParameterDirection

(Optional) The direction type of the parameter.

The default is ParameterDirection.Input.

Returns

Appeon.Data.ParamValue

Returns the newly created ParamValue object.

Examples

创建一个ParamValue对象，并为其指定name及Direction, 并指定数据类型以这种方式创建时，建议Direction=output

using Appeon.ApiDoc.Models.School;
using Appeon.Data;
using System;
using System.Data;

namespace Appeon.ApiDoc.ParamValueExamples
{
    public class NewExample
    {
        private readonly SchoolContext _context;

        public NewExample(SchoolContext context)
        {
            _context = context;
        }

        public void Example3()
        {
            // 创建存放string类型数据的ParamValue对象
            // 并指定其为output参数
            var outParam = ParamValue.New<string>("data1",
                ParameterDirection.Output);

            // 创建一个input类型的ParamValue对象
            var inParam = ParamValue.New<int>("id", 1,
                ParameterDirection.Input);

            // 输出outParam的name，value及Direction信息
            Console.WriteLine("ParamValue.Name = {0}", outParam.Name);
            Console.WriteLine("ParamValue.Value = {0}", outParam.Value);
            Console.WriteLine("ParamValue.DataType = {0}", outParam.DataType);
            Console.WriteLine("ParamValue.Direction = {0}", outParam.Direction);

            // 指定要执行的procedure
            string procedure = "select_course";

            /*
                         
             select_course sql statements:
             update Course set Title = 'English' where CourseID = @id;
             select output = 'success';
             */

            // 使用创建的ParamValue对象，执行procedure
            var row = _context.SqlExecutor.ExecuteProcedure(procedure,
                inParam, outParam);

            // 再次输出outParam的value信息
            Console.WriteLine();
            Console.WriteLine("ParamValue.Value = {0}", outParam.Value);

            /*This code produces the following output:
            
            ParamValue.Name = data1
            ParamValue.Value =
            ParamValue.DataType = System.String
            ParamValue.Direction = Output

            ParamValue.Value = success
            */
        }
    }
}
